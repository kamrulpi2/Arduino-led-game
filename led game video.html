const int TRIG_PINS[] = {6, 9, 12};   // Arduino pins connected to Ultrasonic Sensors' TRIG pins const int ECHO_PINS[] = {7, 10, 13};  // Arduino pins connected to Ultrasonic Sensors' ECHO pins const int BUZZER_PINS[] = {3, 4, 5}; // Arduino pins connected to Piezo Buzzers' pins const int DISTANCE_THRESHOLD = 50;    // centimeters // variables will change: float duration_us, distance_cm; void setup() {   Serial.begin (9600); // initialize serial port   for (int i = 0; i < 3; i++) {     pinMode(TRIG_PINS[i], OUTPUT); // set TRIG pins to output mode     pinMode(ECHO_PINS[i], INPUT);  // set ECHO pins to input mode     pinMode(BUZZER_PINS[i], OUTPUT); // set Buzzer pins to output mode   } } void loop() {   for (int i = 0; i < 3; i++) {     // generate 10-microsecond pulse to TRIG pin of each sensor     digitalWrite(TRIG_PINS[i], HIGH);     delayMicroseconds(10);     digitalWrite(TRIG_PINS[i], LOW);     // measure duration of pulse from ECHO pin of each sensor     duration_us = pulseIn(ECHO_PINS[i], HIGH);     // calculate the distance for each sensor     distance_cm = 0.017 * duration_us;     // Control buzzer based on distance for each sensor     if (distance_cm < DISTANCE_THRESHOLD) {       digitalWrite(BUZZER_PINS[i], HIGH); // turn on Piezo Buzzer for this sensor     } else {       digitalWrite(BUZZER_PINS[i], LOW); // turn off Piezo Buzzer for this sensor     }     // print the value to Serial Monitor     Serial.print("Sensor ");     Serial.print(i + 1);     Serial.print(" distance: ");     Serial.print(distance_cm);     Serial.println(" cm");          delay(10); // Delay between each sensor reading (if needed)   } }
